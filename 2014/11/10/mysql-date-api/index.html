mysql date time datetime类型操作api

返回当前时间

CURDATE() Return the current date

SELECT CURDATE();
> 2014-11-10
SELECT CURDATE() + 0;
> 20141110

CURRENT_DATE(), CURRENT_DATE Synonyms for CURDATE()

select CURRENT_DATE();
> 2014-11-10
select CURRENT_DATE;
> 2014-11-10

CURRENT_TIME(), CURRENT_TIME Synonyms for CURTIME()

select CURRENT_TIME();
> 20:10:41
select CURRENT_TIME;
> 20:10:53

CURRENT_TIMESTAMP(), CURRENT_TIMESTAMP Synonyms for NOW()

select CURRENT_TIMESTAMP();
> 2014-11-10 20:11:10

select CURRENT_TIMESTAMP

CURTIME() Return the current time

select CURTIME()
> 20:11:49
select CURTIME() + 0
> 201205

LOCALTIME(), LOCALTIME Synonym for NOW()

LOCALTIMESTAMP, LOCALTIMESTAMP() Synonym for NOW()

NOW() Return the current date and time

SELECT NOW();
> 2014-11-10 20:46:43

SELECT NOW() + 0;
> 20141110204659

SELECT NOW(), SLEEP(2), NOW();
> 2014-11-10 20:47:25    0    2014-11-10 20:47:25

SELECT SYSDATE(), SLEEP(2), SYSDATE();
> 2014-11-10 20:48:03    0    2014-11-10 20:48:05

SYSDATE() Return the time at which the function executes

SELECT SYSDATE(), SLEEP(2), SYSDATE();
> 2014-11-10 20:58:11    0    2014-11-10 20:58:13

UTC_DATE() Return the current UTC date

SELECT UTC_DATE(), UTC_DATE() + 0;
> 2014-11-10    20141110

SELECT UTC_TIME(), UTC_TIME() + 0;

UTC_TIME()    Return the current UTC time
> 13:10:07    131007

UTC_TIMESTAMP() Return the current UTC date and time

SELECT UTC_TIMESTAMP(), UTC_TIMESTAMP() + 0;
> 2014-11-10 13:10:35    20141110131035

获取指定字段

YEAR() Return the year

SELECT YEAR('1987-01-01');
> 1987

MONTH() Return the month from the date passed

SELECT MONTH('2008-02-03');
> 2

DAY() Synonym for DAYOFMONTH()

DAYOFMONTH() Return the day of the month (0-31)

SELECT DAYOFMONTH('2007-02-03');
> 3

HOUR() Extract the hour

SELECT HOUR('10:05:03');
> 10

SELECT HOUR('272:59:59');
> 272

MINUTE() Return the minute from the argument

SELECT MINUTE('2008-02-03 10:05:03');
> 5

SECOND() Return the second (0-59)

SELECT SECOND('10:05:03');
> 3

DAYNAME() Return the name of the weekday

SELECT DAYNAME('2007-02-03');
> Saturday

DAYOFWEEK() Return the weekday index of the argument

SELECT DAYOFWEEK('2007-02-03');
> 7

DAYOFYEAR() Return the day of the year (1-366)

SELECT DAYOFYEAR('2007-02-03');
> 34

EXTRACT() Extract part of a date

SELECT EXTRACT(YEAR FROM '2009-07-02');
> 2009

SELECT EXTRACT(YEAR_MONTH FROM '2009-07-02 01:02:03');
> 200907

SELECT EXTRACT(DAY_MINUTE FROM '2009-07-02 01:02:03');
> 20102

SELECT EXTRACT(MICROSECOND FROM '2003-01-02 10:30:00.000123');
> 123

LAST_DAY Return the last day of the month for the argument

SELECT LAST_DAY('2003-02-05');
> '2003-02-28'

SELECT LAST_DAY('2004-02-05');
> '2004-02-29'

SELECT LAST_DAY('2004-01-01 01:01:01');
> '2004-01-31'

SELECT LAST_DAY('2003-03-32');
> NULL

MAKEDATE() Create a date from the year and day of year

SELECT MAKEDATE(2011,31), MAKEDATE(2011,32);
> '2011-01-31', '2011-02-01'

SELECT MAKEDATE(2011,365), MAKEDATE(2014,365);
> '2011-12-31', '2014-12-31'

SELECT MAKEDATE(2011,0);
> NULL

MAKETIME MAKETIME()

SELECT MAKETIME(12,15,30);
> 12:15:30

MICROSECOND() Return the microseconds from argument

SELECT MICROSECOND('12:00:00.123456');
> 123456

SELECT MICROSECOND('2009-12-31 23:59:59.000010');
> 10

MONTHNAME() Return the name of the month

SELECT MONTHNAME('2008-02-03');
> February

QUARTER() Return the quarter from a date argument

SELECT QUARTER('2008-04-01');
> 2

TIME() Extract the time portion of the expression passed

SELECT TIME('2003-12-31 01:02:03');
> 01:02:03

SELECT TIME('2003-12-31 01:02:03.000123');
> 01:02:03.000123

TIMESTAMP() With a single argument, this function returns the date or datetime expression; with two arguments, the sum of the arguments

SELECT TIMESTAMP('2003-12-31');
> 2003-12-31 00:00:00

SELECT TIMESTAMP('2003-12-31 12:00:00','12:00:00');
> 2004-01-01 00:00:00

UNIX_TIMESTAMP() Return a UNIX timestamp

SELECT UNIX_TIMESTAMP();
> 1415624876

SELECT UNIX_TIMESTAMP('2007-11-30 10:30:19');
> 1196389819

SELECT FROM_UNIXTIME(1111885200);
> 2005-03-27 09:00:00

WEEK() Return the week number

SELECT WEEK('2008-02-20');
> 7
SELECT WEEK('2008-02-20',0);
> 7

SELECT WEEK('2008-02-20',1);
> 8
SELECT WEEK('2008-12-31',1);
> 53

SELECT YEAR('2000-01-01'), WEEK('2000-01-01',0);
> 2000    0

WEEKDAY() Return the weekday index

SELECT WEEKDAY('2008-02-03 22:23:00');
> 6
SELECT WEEKDAY('2007-11-06');
> 1

WEEKOFYEAR() Return the calendar week of the date (0-53)

SELECT WEEKOFYEAR('2008-02-20');
> 8

YEARWEEK() Return the year and week

SELECT YEARWEEK('1987-01-01');
> 198652

时间相加减计算

ADDDATE() Add time values (intervals) to a date value

SELECT ADDDATE('2008-01-02', INTERVAL 31 DAY); 按天计算
> 2008-02-02
SELECT ADDDATE('2008-01-02', 31);
> 2008-02-02

ADDTIME() Add time

SELECT ADDTIME('2007-12-31 23:59:59.999999', '1 1:1:1.000002'); 
> 2008-01-02 01:01:01.000001

SELECT ADDTIME('01:00:00.999999', '02:00:00.999998');
> 03:00:01.999997

方法语法

DATE_ADD(date,INTERVAL expr unit), DATE_SUB(date,INTERVAL expr unit)

可选值列表

unit Value Expected expr Format

SECOND SECONDS
MINUTE MINUTES
HOUR HOURS
DAY DAYS
WEEK WEEKS
MONTH MONTHS
QUARTER QUARTERS
YEAR YEARS
SECOND_MICROSECOND ‘SECONDS.MICROSECONDS’
MINUTE_MICROSECOND ‘MINUTES:SECONDS.MICROSECONDS’
MINUTE_SECOND ‘MINUTES:SECONDS’
HOUR_MICROSECOND ‘HOURS:MINUTES:SECONDS.MICROSECONDS’
HOUR_SECOND ‘HOURS:MINUTES:SECONDS’
HOUR_MINUTE ‘HOURS:MINUTES’
DAY_MICROSECOND ‘DAYS HOURS:MINUTES:SECONDS.MICROSECONDS’
DAY_SECOND ‘DAYS HOURS:MINUTES:SECONDS’
DAY_MINUTE ‘DAYS HOURS:MINUTES’
DAY_HOUR ‘DAYS HOURS’
YEAR_MONTH ‘YEARS-MONTHS’ 
SELECT DATE_ADD('2008-01-02', INTERVAL 31 DAY);
> 2008-02-02

SELECT '2008-12-31 23:59:59' + INTERVAL 1 SECOND;
> 2009-01-01 00:00:00

SELECT INTERVAL 1 DAY + '2008-12-31';
> 2009-01-01

SELECT DATE_ADD('2000-12-31 23:59:59',INTERVAL 1 SECOND);
> 2001-01-01 00:00:00

SELECT DATE_ADD('2010-12-31 23:59:59', INTERVAL 1 DAY);
> 2011-01-01 23:59:59

SELECT DATE_ADD('2100-12-31 23:59:59', INTERVAL '1:1' MINUTE_SECOND);
> 2101-01-01 00:01:00

SELECT DATE_SUB('2005-01-01 00:00:00', INTERVAL '1 1:1:1' DAY_SECOND);
> 2004-12-30 22:58:59

SELECT DATE_ADD('1900-01-01 00:00:00', INTERVAL '-1 10' DAY_HOUR);
> 1899-12-30 14:00:00

SELECT DATE_SUB('1998-01-02', INTERVAL 31 DAY);
> 1997-12-02

SELECT DATE_ADD('1992-12-31 23:59:59.000002', INTERVAL '1.999999' SECOND_MICROSECOND);
> 1993-01-01 00:00:01.000001

SELECT 6/4;
> 1.5000

SELECT DATE_ADD('2009-01-01', INTERVAL 6/4 HOUR_MINUTE); 增加1小时 5000分钟
> 2009-01-04 12:20:00

SELECT DATE_ADD('2009-01-01', INTERVAL 1.5 HOUR_MINUTE); 增加1小时5分钟
> 2009-01-01 01:05:00

SELECT CAST(6/4 AS DECIMAL(3,1));
> 1.5
SELECT DATE_ADD('1970-01-01 12:00:00', INTERVAL CAST(6/4 AS DECIMAL(3,1)) HOUR_MINUTE);
> 1970-01-01 13:05:00

SELECT DATE_ADD('2013-01-01', INTERVAL 1 DAY);
> 2013-01-02

SELECT DATE_ADD('2013-01-01', INTERVAL 1 HOUR);
> 2013-01-01 01:00:00

SELECT DATE_ADD('2009-01-30', INTERVAL 1 MONTH);
> 2009-02-28

SELECT DATE_ADD('2006-07-00', INTERVAL 1 DAY);
> NULL

SELECT '2005-03-32' + INTERVAL 1 MONTH;
> NULL

DATE_SUB() Subtract a time value (interval) from a date

select DATE_SUB('2014-11-11',INTERVAL 3 DAY);
> 2014-10-08

select DATE_SUB('2014-11-11',INTERVAL 30 DAY);
> 2014-10-12

select date_sub('2014-01-01',interval 31 day);
> 2013-12-01

DATE() Extract the date part of a date or datetime expression

SELECT DATE('2003-12-31 01:02:03');
> 2003-12-31

DATEDIFF()    Subtract two dates
SELECT DATEDIFF('2007-12-31 23:59:59','2007-12-30');
> 1

select datediff('2014-11-11','2014-11-12');
> -1

select datediff('2014-11-12','2014-11-12');
> 0

select datediff('2014-11-12','2014-11-10');
> 2

select datediff('2014-11-12','2014-11-10 13:00');
> 2

select datediff('2014-11-12','2014-11-10 11:00');
> 2

select datediff('2014-11-12','2014-11-10 00:00:01');
> 2

SELECT DATEDIFF('2010-11-30 23:59:59','2010-12-31');
> -31

SUBDATE() A synonym for DATE_SUB() when invoked with three arguments

SELECT DATE_SUB('2008-01-02', INTERVAL 31 DAY);
> 2007-12-02

SELECT SUBDATE('2008-01-02', INTERVAL 31 DAY);
> 2007-12-02

SELECT SUBDATE('2008-01-02 12:00:00', 31);
> 2007-12-02 12:00:00

SUBTIME()    Subtract times
SELECT SUBTIME('2007-12-31 23:59:59.999999','1 1:1:1.000002');
> 2007-12-30 22:58:58.999997

SELECT SUBTIME('01:00:00.999999', '02:00:00.999998');
> -00:59:59.999999

TIMEDIFF() Subtract time

SELECT TIMEDIFF('2000:01:01 00:00:00', '2000:01:01 00:00:00.000001');
> -00:00:00.000001

SELECT TIMEDIFF('2008-12-31 23:59:59.000001', '2008-12-30 01:01:01.000002');
> 46:58:57.999999

TIMESTAMPADD() Add an interval to a datetime expression

SELECT TIMESTAMPADD(MINUTE,1,'2003-01-02');
> 2003-01-02 00:01:00

SELECT TIMESTAMPADD(WEEK,1,'2003-01-02');
> 2003-01-09

TIMESTAMPDIFF() Subtract an interval from a datetime expression

SELECT TIMESTAMPDIFF(MONTH,'2003-02-01','2003-05-01');
> 3

SELECT TIMESTAMPDIFF(YEAR,'2002-05-01','2001-01-01');
> -1

SELECT TIMESTAMPDIFF(MINUTE,'2003-02-01','2003-05-01 12:05:55');
> 128885

PERIOD_ADD() Add a period to a year-month YYYYMM

SELECT PERIOD_ADD(200801,2);
> 200803

PERIOD_DIFF() Return the number of months between periods

SELECT PERIOD_DIFF(200802,200703);
> 11

时间格式化

DATE_FORMAT() Format date as specified

Specifier Description
%a Abbreviated weekday name (Sun..Sat)
%b Abbreviated month name (Jan..Dec)
%c Month, numeric (0..12)
%D Day of the month with English suffix (0th, 1st, 2nd, 3rd, …)
%d Day of the month, numeric (00..31)
%e Day of the month, numeric (0..31)
%f Microseconds (000000..999999)
%H Hour (00..23)
%h Hour (01..12)
%I Hour (01..12)
%i Minutes, numeric (00..59)
%j Day of year (001..366)
%k Hour (0..23)
%l Hour (1..12)
%M Month name (January..December)
%m Month, numeric (00..12)
%p AM or PM
%r Time, 12-hour (hh:mm:ss followed by AM or PM)
%S Seconds (00..59)
%s Seconds (00..59)
%T Time, 24-hour (hh:mm:ss)
%U Week (00..53), where Sunday is the first day of the week
%u Week (00..53), where Monday is the first day of the week
%V Week (01..53), where Sunday is the first day of the week; used with %X
%v Week (01..53), where Monday is the first day of the week; used with %x
%W Weekday name (Sunday..Saturday)
%w Day of the week (0=Sunday..6=Saturday)
%X Year for the week where Sunday is the first day of the week, numeric, four digits; used with %V
%x Year for the week, where Monday is the first day of the week, numeric, four digits; used with %v
%Y Year, numeric, four digits
%y Year, numeric (two digits)
%% A literal “%” character
%x x, for any “x” not listed above 
SELECT DATE_FORMAT('2009-10-04 22:23:00', '%W %M %Y');
> Sunday October 2009

SELECT DATE_FORMAT('2007-10-04 22:23:00', '%H:%i:%s');
> 22:23:00

SELECT DATE_FORMAT('1900-10-04 22:23:00', '%D %y %a %d %m %b %j');
> 4th 00 Thu 04 10 Oct 277

SELECT DATE_FORMAT('1997-10-04 22:23:00', '%H %k %I %r %T %S %w');
> 22 22 10 10:23:00 PM 22:23:00 00 6

SELECT DATE_FORMAT('1999-01-01', '%X %V'); 
> 1998 52

SELECT DATE_FORMAT('2006-06-00', '%d');
> 00

和字符串之间转换

STR_TO_DATE() Convert a string to a date

SELECT STR_TO_DATE('01,5,2013','%d,%m,%Y');
> 2013-05-01

SELECT STR_TO_DATE('May 1, 2013','%M %d,%Y');
> '2013-05-01'

SELECT STR_TO_DATE('a09:30:17','a%h:%i:%s');
> 09:30:17

SELECT STR_TO_DATE('a09:30:17','%h:%i:%s');
> NULL

SELECT STR_TO_DATE('09:30:17a','%h:%i:%s');
> 09:30:17

SELECT STR_TO_DATE('abc','abc');
> 0000-00-00

SELECT STR_TO_DATE('9','%m');
> 0000-09-00

SELECT STR_TO_DATE('9','%s');
> 00:00:09

SELECT STR_TO_DATE('00/00/0000', '%m/%d/%Y');
> 0000-00-00

SELECT STR_TO_DATE('04/31/2004', '%m/%d/%Y');
> 2004-04-31

SELECT STR_TO_DATE('200442 Monday', '%X%V %W');
> 2004-10-18

TIME_FORMAT() Format as time

SELECT TIME_FORMAT('100:00:00', '%H %k %h %I %l');
